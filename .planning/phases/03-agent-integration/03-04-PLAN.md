  healthEndpoint: '/health'

**Save handler:**
```
const handleSave = async () => {
  if (!name.trim()) return;
  if (transportType === 'stdio' && !command.trim()) return;
  if (transportType === 'streamable-http' && !url.trim()) return;

  const config = transportType === 'stdio'
    ? { type: 'stdio', command: command.trim(), args: args.split(',').map(a => a.trim()).filter(Boolean) }
    : { type: 'streamable-http', url: url.trim(), healthEndpoint: healthEndpoint.trim() || '/health' };

  const id = server?.id || crypto.randomUUID();
  await saveMcpServer({ id, name: name.trim(), server_type: transportType, config });
  onSaved();
};
```
