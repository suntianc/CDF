function getAgentMcpServers(agentId: string): MCPServer[] {
  const rows = db.prepare(`
    SELECT m.*
    FROM mcp_servers m
    INNER JOIN agent_mcp_servers ams ON ams.mcp_server_id = m.id
    WHERE ams.agent_id = ?
    ORDER BY m.updated_at DESC
  `).all(agentId) as Array<...>;
  return rows.map(...);
}
