       function startWatcher(
         dir: string,
         onChange: () => Promise<void>,
         scope: 'system' | 'project'
       ): () => void {
         let degraded = false;
         let handle: chokidar.FSWatcher | null = null;
         try {
           handle = chokidar.watch(dir, {
             awaitWriteFinish: { stabilityThreshold: 200, pollInterval: 100 },
             ignoreInitial: true,
             depth: 0,
             usePolling: false,
           });
         } catch (err) {
           log.error('[commands-watcher] chokidar watch threw on construction:', err);
           void degradeAndFallback(err as Error, dir, onChange, scope);
           return () => {};
         }

         const fire = debounce(async () => {
           try {
             await onChange();
           } catch (err) {
             log.error('[commands-watcher] onChange callback failed:', err);
           }
           BrowserWindow.getAllWindows().forEach((w) => {
             w.webContents.send('commands:changed', { source: 'chokidar' });
           });
         }, 100);

         ['add', 'change', 'unlink'].forEach((evt) => {
           handle!.on(evt, () => {
             void fire();
           });
         });

         handle.on('error', (err) => {
           log.error('[commands-watcher] chokidar error:', err);
           if (!degraded) {
             degraded = true;  // D-19: first-error wins, no retry
             void handle?.close();
             handle = null;
             void degradeAndFallback(err, dir, onChange, scope);
           }
         });

         return () => {
           void handle?.close();
         };
       }

       async function degradeAndFallback(
         err: Error,
         dir: string,
         onChange: () => Promise<void>,
         scope: 'system' | 'project'
       ): Promise<void> {

<threat_model>
## Trust Boundaries
| Boundary | Description |
|----------|-------------|
| main → renderer (commands:fallback IPC) | Main pushes {error, scope} payload to all windows. ASVS V11: validate scope is 'system' or 'project'; renderer should not trust arbitrary payload. |
| renderer → sonner (toast.warning) | toast.description interpolates scope + error. sonner 2.0.7 default-escapes. |
| chokidar → main (error event) | chokidar's error event may contain raw OS error (path, errno). No path traversal risk (paths are anchored to ~/.cdf/commands or <project>/.cdf/commands). |

## STRIDE Threat Register
| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-T6 | Denial of Service | Repeated chokidar 'error' events | mitigate | `degraded` flag (D-19): first-error wins, subsequent 'error' events are no-op. |
| T-08-T7 | Information Disclosure | toast.description with error message | mitigate | sonner default-escapes; error messages are chokidar's standard (e.g. "EACCES: permission denied") — no PII. |
| T-08-T8 | Spoofing | onFallback with arbitrary scope string | accept | Type signature enforces 'system' | 'project'; renderer reads data.scope without further validation (low risk: scope only controls display label). |
| T-08-T9 | Tampering | toastedFingerprints Set | accept | In-memory Set, no persistence, no security boundary. |
| T-08-SC | Tampering | npm installs | mitigate | No new packages; chokidar + sonner already installed. |
