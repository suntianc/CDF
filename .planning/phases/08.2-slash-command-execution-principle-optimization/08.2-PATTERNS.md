```typescript
ipcMain.handle('commands:readBody', async (_evt, bodyPath: string) => {
  try {
    // ASVS V5.1.3 input validation — bodyPath must be a string under an allowlisted dir
    if (typeof bodyPath !== 'string' || bodyPath.length === 0 || bodyPath.length > 1024) {
      return { body: '', mtimeMs: 0 };
    }
    const allowedPrefixes = [
      path.join(os.homedir(), '.cdf', 'commands'),
      // Project paths resolved at call-time via the registry's known project paths
    ];
    const resolved = path.resolve(bodyPath);
    const isAllowed = allowedPrefixes.some((p) => resolved.startsWith(p + path.sep));
    if (!isAllowed) {
      console.warn('[commands:readBody] path not under allowed dir:', bodyPath);
      return { body: '', mtimeMs: 0 };
    }
    if (!fs.existsSync(resolved)) {
      return { body: '', mtimeMs: 0 };
    }
    const stat = fs.statSync(resolved);
    const body = fs.readFileSync(resolved, 'utf-8');
    return { body, mtimeMs: stat.mtimeMs };
  } catch (err) {
    console.error('[commands:readBody] failed:', err);
    return { body: '', mtimeMs: 0 };
  }
});
```

- Path traversal guard: must verify `bodyPath` resolves under `~/.cdf/commands/` (or a project's `.cdf/commands/`). Phase 6 list-system-context resolution at `src/main/commands/project-commands.ts:71-79` is the model.

### `src/main/commands/project-commands.ts` (EXTEND parser) — main-parser, file-I/O + parse

**Current parser (lines 27-46):**
```typescript
export function parseFrontmatter(filePath: string): CommandFrontmatter {
  if (!fs.existsSync(filePath)) return {};
  const content = fs.readFileSync(filePath, 'utf-8');
  if (!content.startsWith(FRONTMATTER_START)) return {};

  const end = content.indexOf(FRONTMATTER_END, FRONTMATTER_START.length);
  if (end === -1) return {};

  const body = content.slice(FRONTMATTER_START.length, end);
  const result: CommandFrontmatter = {};
  for (const line of body.split('\n')) {
    const colon = line.indexOf(':');
    if (colon === -1) continue;
    const key = line.slice(0, colon).trim();
    const value = line.slice(colon + 1).trim();
    if (!key) continue;
    (result as Record<string, string>)[key] = value;
  }
  return result;
}
```

**08.2 D-08 replacement (uses `yaml@2.9.0`):**

```typescript
import YAML from 'yaml';

export function parseFrontmatter(filePath: string): ParsedFrontmatter {
  if (!fs.existsSync(filePath)) return {};
  const content = fs.readFileSync(filePath, 'utf-8');
  if (!content.startsWith(FRONTMATTER_START)) return {};

  const end = content.indexOf(FRONTMATTER_END, FRONTMATTER_START.length);
  if (end === -1) return {};

  const raw = content.slice(FRONTMATTER_START.length, end);
  // yaml@2.9.0 parse — returns { key: typedValue } where booleans/arrays/strings are correctly typed
  const parsed = YAML.parse(raw) ?? {};
  return {
    disableModelInvocation: parsed['disable-model-invocation'],
    userInvocable: parsed['user-invocable'] ?? true,
    allowedTools: parsed['allowed-tools'] ?? [],
    whenToUse: parsed['when_to_use'] ?? '',
    arguments: parsed['arguments'] ?? [],
  };
}
```

### `src/renderer/src/lib/commands/argSubstitution.ts` (NEW) — renderer-util, pure string transform

**Analog:** `src/renderer/src/lib/commands/parseInputToTokens.ts` (existing pure-token transform)

**Why this analog:** `parseInputToTokens.ts` is a small, pure TypeScript module that exports testable functions and has a co-located `.test.ts` file. The exact same shape fits `argSubstitution.ts`.

**Signature (RESEARCH.md §"Pattern: Lazy body loading on dispatch"):**
```typescript
/**
 * Pure string substitution for /command body placeholders.
 * - $ARGUMENTS → entire args string verbatim
 * - $0 / $1 / $2 / ... → positional args (whitespace-split)
 * - $name → named arg (frontmatter `arguments: [name1, name2]` declaration)
 * - No eval, no flag parsing — raw substitution per D-02 / P7 pitfall
 */
export interface ArgSubstitutionContext {
  args: string;            // e.g. "production --force 顺带更新 changelog"
  arguments?: string[];    // frontmatter `arguments: [env, flags]` — names only
  namedArgs?: Record<string, string>;  // optional, populated by future parser (D-02 future)
}

export function substituteArgs(body: string, ctx: ArgSubstitutionContext): string {
  // Split args on whitespace for positional lookup
  const positional = ctx.args.trim().split(/\s+/).filter(Boolean);

  // 1) $ARGUMENTS (whole args, verbatim) — D-03 the body replaces the user message
  let out = body.replaceAll('$ARGUMENTS', ctx.args);

  // 2) $0, $1, $2, ... — positional; $0 is the first arg
  out = out.replace(/\$(\d+)/g, (m, idx) => {
    const i = parseInt(idx, 10);
    return positional[i] ?? m;  // leave placeholder intact if out-of-range
  });

  // 3) $name — named arg (declared in frontmatter `arguments:`)
  if (ctx.arguments && ctx.arguments.length > 0) {
    const named: Record<string, string> = {};
    ctx.arguments.forEach((name, i) => {
      named[name] = positional[i] ?? '';
    });
    out = out.replace(/\$([a-zA-Z_][a-zA-Z0-9_]*)/g, (m, name) => {
      if (name in named) return named[name];
      return m;  // leave placeholder intact if undeclared
    });
  }
  return out;
}
```

**Pattern notes:**
- **NO `eval`**, no flag parsing — pure `.replaceAll()` per PITFALLS P7 + D-02 (CONTEXT.md).
- `$ARGUMENTS` is the **only** "all-args" placeholder; `$0..$N` are positional.
- The `arguments: [env, flags]` frontmatter declaration is what gates `$name` lookup (security boundary).
- Co-located test: `argSubstitution.test.ts` mirrors `formatTokenLabel.test.ts` structure.
