Create `src/renderer/src/hooks/useStreaming.ts` with the following:

**Interface:**
```typescript
interface UseStreamingOptions {
  sessionPath: string
  onToken?: (token: string) => void
  onEnd?: () => void
  onError?: (error: string) => void
}

interface UseStreamingReturn {
  streamId: string | null
  isStreaming: boolean
  start: () => void
  stop: () => void
}
```

**Implementation:**
1. Call `window.api.session.startStream(sessionPath)` to get streamId
2. Subscribe to tokens via `window.api.session.onStreamToken(streamId, (data) => {...})`
3. On token: call onToken callback AND update store via `useMessageStore.getState().appendContent(messageId, delta)`
4. On end: call onEnd, update store via `finalizeMessage(messageId)`
5. On error: call onError, update message status to 'error'
6. Return `{ streamId, isStreaming, start, stop }`

**Key details:**
- Track messageId for the assistant message being streamed (passed via options or stored)
- Use `useMessageStore.getState()` for direct store access (not hook during streaming to avoid re-renders)
- Return cleanup function from onStreamToken subscription
- `stop()` calls `window.api.session.stopStream(sessionPath)` and cleans up subscription

```typescript
export function useStreaming(options: UseStreamingOptions): UseStreamingReturn {
  const { sessionPath, onToken, onEnd, onError, messageId } = options
  const [streamId, setStreamId] = useState<string | null>(null)
  const [isStreaming, setIsStreaming] = useState(false)
  const unsubscribeRef = useRef<(() => void) | null>(null)

  const start = useCallback(() => {
    if (isStreaming) return
    const id = window.api.session.startStream(sessionPath)
    setStreamId(id)
    setIsStreaming(true)

    unsubscribeRef.current = window.api.session.onStreamToken(id, (data) => {
      if (data.type === 'token' && data.delta) {
        onToken?.(data.delta)
        useMessageStore.getState().appendContent(messageId, data.delta)
      } else if (data.type === 'end') {
        useMessageStore.getState().finalizeMessage(messageId)
        onEnd?.()
        setIsStreaming(false)
      } else if (data.type === 'error') {
        useMessageStore.getState().updateMessageStatus(messageId, 'error')
        onError?.(data.message || 'Stream error')
        setIsStreaming(false)
      }
    })
  }, [sessionPath, messageId, isStreaming, onToken, onEnd, onError])

  const stop = useCallback(() => {
    window.api.session.stopStream(sessionPath)
    unsubscribeRef.current?.()
    setIsStreaming(false)
  }, [sessionPath])

  // Cleanup on unmount
  useEffect(() => {
    return () => {
      unsubscribeRef.current?.()
      if (isStreaming) {
        window.api.session.stopStream(sessionPath)
      }
    }
  }, [])

  return { streamId, isStreaming, start, stop }
}
```

components={{
  code({ node, className, children, ...props }) {
    const match = /language-(\w+)/.exec(className || '')
    const code = String(children).replace(/\n$/, '')
    
    if (match && code) {
      // Use shiki for syntax highlighting
      const html = await codeToHtml(code, {
        lang: match[1],
        theme: 'github-dark'
      })
      return <code dangerouslySetInnerHTML={{ __html: html }} {...props} />

    // Inline code (no language)
