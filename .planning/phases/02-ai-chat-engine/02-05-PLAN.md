function formatRelativeTime(isoString: string): string {
  const date = new Date(isoString)
  const now = new Date()
  const diff = now.getTime() - date.getTime()
  const minutes = Math.floor(diff / 60000)
  const hours = Math.floor(diff / 3600000)
  const days = Math.floor(diff / 86400000)
  
  if (minutes < 1) return '刚刚'
  if (minutes < 60) return `${minutes}分钟前`
  if (hours < 24) return `${hours}小时前`
  if (days < 7) return `${days}天前`
  return date.toLocaleDateString('zh-CN')
}

// New conversation handler
const handleNewConversation = async () => {
  try {
    const { id, path } = await window.api.chatHistory.create(workspacePath)
    setActiveConversationId(id)
    setConversationPath(path)
    useMessageStore.getState().clearAll()
    useMessageStore.getState().setConversation(id)
  } catch (err) {
    console.error('Failed to create conversation:', err)
  }
}

// Select conversation handler
const handleSelectConversation = async (id: string, path: string) => {
  try {
    setActiveConversationId(id)
    setConversationPath(path)
    const messages = await window.api.chatHistory.load(path)
    useMessageStore.getState().loadHistory(messages)
    useMessageStore.getState().setConversation(id)
  } catch (err) {
    console.error('Failed to load conversation:', err)
  }
}

// Clear conversation handler
const handleClearConversation = async () => {
  useMessageStore.getState().clearAll()
  if (conversationPath) {
    await window.api.chatHistory.save(conversationPath, [])
  }
}
