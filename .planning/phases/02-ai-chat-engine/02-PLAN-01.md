### Task 1.3: Implement streaming event forwarding (main → renderer)

<read_first>
- `.planning/phases/02-ai-chat-engine/02-RESEARCH-SUPPLEMENT.md` (Section 2: Stream event protocol)
- `src/main/ipc.ts` (existing IPC handler pattern)
</read_first>

<action>
Add streaming IPC infrastructure. This sends real-time AI response chunks from main to renderer.

1. Add a streaming handler that receives a session path and sends events:

```typescript
// In src/main/ipc.ts
ipcMain.on('session:startStream', async (event, { sessionPath }) => {
  const { SessionManager } = await import('@earendil-works/pi-coding-agent');
  const session = SessionManager.open(sessionPath);
  const browserWindow = BrowserWindow.fromWebContents(event.sender);

  try {
    // Build context with current session state
    const context = session.buildSessionContext();

    // Use pi SDK agent to generate streaming response
    // NOTE: agent setup will be refined in Plan 03 — this is the streaming foundation
    const stream = await agent.generateStream(context.messages, context.settings);

    for await (const chunk of stream) {
      if (browserWindow && !browserWindow.isDestroyed()) {
        event.sender.send('session:streamChunk', {
          type: chunk.type,    // 'text' | 'thinking' | 'toolCall' | 'toolResult'
          content: chunk.content,
          metadata: { timestamp: new Date().toISOString() }
        });
      }
    }

    // Signal completion
    event.sender.send('session:streamChunk', { type: 'done' });
  } catch (error) {
    event.sender.send('session:streamChunk', {
      type: 'error',
      content: error instanceof Error ? error.message : 'Stream error'
    });
  }
});
```

2. Add stream abort handler:
```typescript
ipcMain.on('session:stopStream', (event) => {
  // Store abort controller reference per session
  // Implementation: use Map<string, AbortController>
  const controller = activeStreams.get(sessionPath);
  if (controller) {
    controller.abort();
    activeStreams.delete(sessionPath);
  }
  event.sender.send('session:streamChunk', { type: 'stopped' });
});
```

3. Create `activeStreams` Map at module scope:
```typescript
const activeStreams = new Map<string, AbortController>();
```

4. Import `BrowserWindow` from `electron` at the top.

5. Add `session:startStream` and `session:stopStream` to the exported register function.
</action>

In `src/preload/index.ts`, extend the existing `contextBridge.exposeInMainWorld()` call:

```typescript
import { contextBridge, ipcRenderer } from 'electron';

contextBridge.exposeInMainWorld('api', {
  // ... existing Phase 1 APIs (store, workspace, theme, providers) ...

  // Session APIs
  session: {
    create: (cwd: string) => ipcRenderer.invoke('session:create', cwd),
    list: (cwd: string) => ipcRenderer.invoke('session:list', cwd),
    open: (path: string) => ipcRenderer.invoke('session:open', path),
    sendMessage: (data: { sessionPath: string; content: string; images?: Array<{ data: string; mimeType: string }> }) =>
      ipcRenderer.invoke('session:sendMessage', data),
    setName: (sessionPath: string, name: string) => ipcRenderer.invoke('session:setName', { sessionPath, name }),
    delete: (sessionPath: string) => ipcRenderer.invoke('session:delete', sessionPath),
    startStream: (sessionPath: string) => ipcRenderer.send('session:startStream', { sessionPath }),
    stopStream: (sessionPath: string) => ipcRenderer.send('session:stopStream', sessionPath),
    onStreamChunk: (callback: (chunk: StreamChunk) => void) => {
      const handler = (_event: any, chunk: StreamChunk) => callback(chunk);
      ipcRenderer.on('session:streamChunk', handler);
      return () => ipcRenderer.removeListener('session:streamChunk', handler);
    }
  },

  // GSD APIs
  gsd: {
    execute: (command: string, args: string[]) => ipcRenderer.invoke('gsd:execute', command, args),
  }
});
```
