export interface Message {
  id: string
  role: 'user' | 'assistant'
  content: string
  timestamp: number
  streaming?: boolean
  status?: 'sending' | 'sent' | 'error'
}

interface MessageStore {
  messages: Message[]
  currentStreamId: string | null
  activeConversationId: string | null
  isStreaming: boolean
  setStreamId: (id: string | null) => void
  setConversation: (id: string | null) => void
  setStreaming: (streaming: boolean) => void
  addMessage: (msg: Omit<Message, 'id' | 'timestamp'>) => string
  appendContent: (id: string, delta: string) => void
  finalizeMessage: (id: string) => void
  updateMessageStatus: (id: string, status: Message['status']) => void
  loadHistory: (messages: Message[]) => void
  clearAll: () => void
}
```

Store must:
- Use `create<MessageStore>((set, get) => ...)` from zustand v5
- `addMessage` generates UUID via `crypto.randomUUID()`, sets timestamp via `Date.now()`
- `appendContent` appends delta to message content by id
- `finalizeMessage` sets streaming=false on message
- `loadHistory` replaces entire messages array (for pagination load)
- `clearAll` resets messages, currentStreamId, activeConversationId, isStreaming
- Export as named export `useMessageStore`

Do NOT use a Provider wrapper - Zustand v5 hooks work without Provider.

Extend `src/main/ipc.ts` with streaming IPC handlers following D-02 (Channel event pattern).

Add the following handlers using the existing pattern from RESEARCH.md:

1. **session:streamStart** (ipcMain.on) - starts pi SDK streaming for a session
   - Generate streamId as `stream-${Date.now()}-${Math.random().toString(36).slice(2)}`
   - Open session via SessionManager.open(sessionPath)
   - Subscribe to session events (pi SDK subscription pattern)
   - On `text_delta` event: send token via `event.sender.send(streamId, { type: 'token', delta })`
   - On `message_end` event: send `{ type: 'end' }`
   - On error: send `{ type: 'error', message: error.message }`
   - Store AbortController in activeStreams Map keyed by sessionPath
   - Return streamId to renderer via the channel

2. **session:streamStop** (ipcMain.on) - aborts active stream
   - Retrieve AbortController from activeStreams by sessionPath
   - Call controller.abort()
   - Remove from activeStreams Map

3. **session:sendMessage** (modify existing) - append user message and initiate stream
   - Append message to session via session.appendMessage
   - Return entryId

// Streaming APIs - extend existing session object
startStream: (sessionPath: string) => {
  const streamId = `stream-${Date.now()}-${Math.random().toString(36).slice(2)}`
  ipcRenderer.send('session:streamStart', { sessionPath, streamId })
  return streamId
},
stopStream: (sessionPath: string) => {
  ipcRenderer.send('session:streamStop', { sessionPath })
},
onStreamToken: (streamId: string, callback: (data: { type: 'token' | 'end' | 'error'; delta?: string; message?: string }) => void) => {
  const handler = (_event: any, data: any) => callback(data)
  ipcRenderer.on(streamId, handler)
  return () => ipcRenderer.removeListener(streamId, handler)
},
```

The renderer will use these to:
1. Call `startStream(sessionPath)` to get streamId
2. Subscribe to tokens via `onStreamToken(streamId, callback)`
3. Call `stopStream(sessionPath)` to abort

Place these in the session API object that is already exposed via contextBridge.
